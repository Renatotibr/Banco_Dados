SELECT COUNT(*) AS TOTAL, P.IDPRJ, P.DESCRICAO FROM MPRJ P
INNER JOIN MTRF T ON T.CODCOLIGADA = P.CODCOLIGADA
AND T.IDPRJ = P.IDPRJ
GROUP BY P.IDPRJ, P.DESCRICAO
HAVING COUNT(*)>1  order by TOTAL desc 

SELECT CODCOLIGADA, CODPACIENTE, CODATENDIMENTO, SEQATEND,COUNT(*) AS QTD 
FROM SZANAMNESECOMPL
GROUP BY CODCOLIGADA, CODPACIENTE, CODATENDIMENTO, SEQATEND
HAVING COUNT(*) >1
